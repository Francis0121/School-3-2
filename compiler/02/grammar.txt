;[주의사항]
;	1.Start symbol is an LHS of the first production rule
;	2. '@'은 epsilon을 의미함.
;
;<참고> 첫 문자가 ';'으로 시작되는 line은 무시함(comment로 간주).
;	 즉, 첫 문자가 ';'가 아닌 line의 생성규칙들만 읽어 들임.
;
;---------------------------------
; CFG 1
;---------------------------------
;E -> a
;E -> E + E
;E -> E - E
;E -> E * E
;E -> E / E
;---------------------------------
; CFG 2
;---------------------------------
S -> bAS
S -> aSA
S -> b
A -> AbS
A -> SaS
A -> a
;
;---------------------------------
; CFG 3
;---------------------------------
;S -> aA
;S -> bB
;S -> @
;A -> aS
;A -> a
;B -> bS
;B -> @
;
;---------------------------------
; CFG 4
;---------------------------------
;A -> aA
;A -> bB
;A -> a
;B -> b
;B -> bB
;
;---------------------------------
; CFG 5
;---------------------------------
;S -> aA0
;S -> bB1
;A -> aA0
;A -> aA
;A -> @
;B -> bB1
;B -> B1
;B -> @
;
;---------------------------------
; CFG 6
;---------------------------------
;S -> abR
;S -> c
;R -> Sba
;R -> d
;
;---------------------------------
; CFG 7
;---------------------------------
;S -> aAb
;S -> Sb
;A -> cS
